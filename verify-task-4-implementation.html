User Name State and Storage Functions

// User name state let userName = ''; function loadUserName() { try { const storedName = localStorage.getItem('userName'); userName = storedName || ''; } catch (e) { console.error('Failed to load user name from localStorage:', e); userName = ''; } } function saveUserName(name) { try { localStorage.setItem('userName', name); userName = name; } catch (e) { console.error('Failed to save user name to localStorage:', e); } }
